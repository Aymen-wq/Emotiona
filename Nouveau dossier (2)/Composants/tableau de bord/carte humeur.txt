  neutral: { bg: 'bg-slate-50', text: 'text-slate-600', accent: '#64748b' },
  low: { bg: 'bg-amber-50', text: 'text-amber-600', accent: '#f59e0b' },
  struggling: { bg: 'bg-rose-50', text: 'text-rose-600', accent: '#f43f5e' }
};

const moodEmojis = {
  excellent: '‚ú®',
  good: 'üòä',
  neutral: 'üòê',
  low: 'üòî',
  struggling: 'üí≠'
};

export default function MoodCard({ mood = 'neutral', emotion = 'Calm', insight, onClick }) {
  const colors = moodColors[mood] || moodColors.neutral;

    <motion.div

      animate={{ opacity: 1, y: 0 }}

          {emotion}
        </h3>
        <p className="text-slate-500 text-sm leading-relaxed">
          {insight || "How are you feeling today?"}
        </p>
      </div>
    </motion.div>

}
